        'stok' => 'required|numeric', 
        'deskripsi' => 'nullable|string', 
        // di modul 4 validasi file-nya adalah 'image'

    ]); 
    
    // Setelah validasi sukses, di sini adalah tempat kode penyimpanan data ke database.
}
